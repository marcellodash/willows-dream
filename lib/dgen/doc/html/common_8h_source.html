260 static inline void Interpolate_2_24(uint24_t *ret, uint24_t c1, int w1, uint24_t c2, int w2, int s)
261 {
262  if (!memcmp(c1, c2, 3)) {
263  u24cpy(ret, c1);
264  return;
265  }
266  (*ret)[0] = (((c1[0] * w1) + (c2[0] * w2)) >> s);
267  (*ret)[1] = (((c1[1] * w1) + (c2[1] * w2)) >> s);
268  (*ret)[2] = (((c1[2] * w1) + (c2[2] * w2)) >> s);
269 }
